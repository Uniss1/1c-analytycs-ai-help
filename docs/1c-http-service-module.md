# Модуль HTTP-сервиса 1С: /analytics_execute и /query

Модуль обработки запросов HTTP-сервиса для AI-ассистента.
Вставить в: Конфигуратор → HTTP-сервисы → `АИАналитика` → Модуль обработки.

## Настройка HTTP-сервиса в конфигураторе

1. Создать HTTP-сервис: `АИАналитика`
2. Корневой URL: `/ai`
3. Шаблоны URL (два):
   - `/analytics_execute` — метод `POST`, обработчик `ОбработатьЗапрос`
   - `/query` — метод `POST`, обработчик `ОбработатьЗапросDiscovery`

Шаблон `/query` нужен только для `scripts/sync_metadata.py`: он принимает
ограниченный whitelist запросов (health-check, probe-регистр, distinct-значения)
и пересобирает текст запроса из доверенного шаблона — сырой БСЛ-текст клиента
никогда не попадает в `Новый Запрос()`. Для регулярных пользовательских
запросов используется только `/analytics_execute`.

// ===========================================================================

// ===========================================================================

#Область ОбработчикиСобытий

Функция ОбработатьЗапрос(Запрос)
	
	Попытка
		ТелоJSON = Запрос.ПолучитьТелоКакСтроку();
		ЧтениеJSON = Новый ЧтениеJSON;
		ЧтениеJSON.УстановитьСтроку(ТелоJSON);
		ДанныеЗапроса = ПрочитатьJSON(ЧтениеJSON);
	Исключение
		Возврат СформироватьОтветОшибки("invalid_params",
			"Некорректный JSON в теле запроса", , 400);
	КонецПопытки;
	
	// Извлечь параметры
	ИмяРегистра = Неопределено;
	ИмяИнструмента = Неопределено;
	Параметры = Неопределено;
	
	Попытка
		ИмяРегистра = ДанныеЗапроса.register;
		ИмяИнструмента = НРег(ДанныеЗапроса.tool);
		Параметры = ДанныеЗапроса.params;
	Исключение
		Возврат СформироватьОтветОшибки("missing_params",
			"Отсутствуют обязательные поля: register, tool, params", , 400);
	КонецПопытки;
	
	// Валидация параметров
	РезультатВалидации = ВалидироватьПараметры(ИмяРегистра, ИмяИнструмента, Параметры);
	Если НЕ РезультатВалидации.Успех Тогда
		Возврат СформироватьОтветОшибки(РезультатВалидации.ТипОшибки,
			РезультатВалидации.Сообщение, РезультатВалидации.ДопустимыеЗначения);
	КонецЕсли;
	
	// Роутинг по инструменту
	Попытка
		Если ИмяИнструмента = "aggregate" Тогда
			Результат = ВыполнитьAggregate(ИмяРегистра, Параметры);
		ИначеЕсли ИмяИнструмента = "group_by" Тогда
			Результат = ВыполнитьGroupBy(ИмяРегистра, Параметры);
		ИначеЕсли ИмяИнструмента = "top_n" Тогда
			Результат = ВыполнитьTopN(ИмяРегистра, Параметры);
		ИначеЕсли ИмяИнструмента = "time_series" Тогда
			Результат = ВыполнитьTimeSeries(ИмяРегистра, Параметры);
		ИначеЕсли ИмяИнструмента = "compare" Тогда
			Результат = ВыполнитьCompare(ИмяРегистра, Параметры);
		ИначеЕсли ИмяИнструмента = "ratio" Тогда
			Результат = ВыполнитьRatio(ИмяРегистра, Параметры);
		ИначеЕсли ИмяИнструмента = "filtered" Тогда
			Результат = ВыполнитьFiltered(ИмяРегистра, Параметры);
		Иначе
			Возврат СформироватьОтветОшибки("invalid_params",
				"Неизвестный инструмент: " + ИмяИнструмента);
		КонецЕсли;
	Исключение
		Возврат СформироватьОтветОшибки("execution_error",
			"Ошибка выполнения запроса: " + ОписаниеОшибки());
	КонецПопытки;
	
	Возврат Результат;

		ТекстЗапросаСырой = СокрЛП(ДанныеЗапроса.query);

		Возврат СформироватьОтветОшибки("missing_params",

	Нормализованный = НормализоватьПробелы(ТекстЗапросаСырой);
	Распознанный = РаспознатьDiscoveryЗапрос(Нормализованный);
	Если Распознанный = Неопределено Тогда
		Возврат СформироватьОтветОшибки("invalid_params",
			"Запрос вне whitelist. Разрешены только health-check, probe-регистр, distinct-значения");
	КонецЕсли;
	
	Попытка
		ТекстБезопасный = СобратьDiscoveryЗапрос(Распознанный);
	Исключение
		Возврат СформироватьОтветОшибки("invalid_params", ОписаниеОшибки());
	КонецПопытки;
	
	Попытка
		ЗапросОбъект = Новый Запрос(ТекстБезопасный);
		РезультатВыполнения = ЗапросОбъект.Выполнить();
		Данные = РезультатВыполненияВМассив(РезультатВыполнения);

		Возврат СформироватьОтветОшибки("execution_error",
			"Ошибка выполнения запроса: " + ОписаниеОшибки());
	КонецПопытки;
	
	Возврат СформироватьОтветУспех(Данные);

// aggregate — одно агрегированное значение за период
Функция ВыполнитьAggregate(ИмяРегистра, Параметры)
	
	Ресурс = Параметры.resource;
	Лимит = ПолучитьЛимит(Параметры);
	
	УсловияОтбора = ПостроитьУсловияОтбора(Параметры);
	
	ТекстЗапроса =
		"ВЫБРАТЬ ПЕРВЫЕ " + Формат(Лимит, "ЧГ=") + "
		|	СУММА(" + Ресурс + ") КАК Значение
		|ИЗ
		|	РегистрСведений." + ИмяРегистра + "
		|" + УсловияОтбора.ТекстГДЕ;
	
	Запрос = Новый Запрос(ТекстЗапроса);
	УстановитьПараметрыОтбора(Запрос, УсловияОтбора.Параметры);
	
	Результат = Запрос.Выполнить();
	Если Результат.Пустой() Тогда
		Возврат СформироватьОтветОшибки("no_data", "Данные за указанный период не найдены");
	КонецЕсли;
	
	Данные = Новый Массив;
	Выборка = Результат.Выбрать();
	Пока Выборка.Следующий() Цикл
		Запись = Новый Структура("Значение", Выборка.Значение);
		Данные.Добавить(Запись);
	КонецЦикла;
	
	Возврат СформироватьОтветУспех(Данные, , ТекстЗапроса);

// group_by — группировка по измерению
Функция ВыполнитьGroupBy(ИмяРегистра, Параметры)
	
	Ресурс = Параметры.resource;
	Лимит = ПолучитьЛимит(Параметры);
	ПоляГруппировки = ПолучитьПоляГруппировки(Параметры);
	Сортировка = ПолучитьСортировку(Параметры);
	
	УсловияОтбора = ПостроитьУсловияОтбора(Параметры);
	
	// Формируем SELECT-поля группировки
	ТекстПолейВыбора = "";
	Для Каждого Поле Из ПоляГруппировки Цикл
		ТекстПолейВыбора = ТекстПолейВыбора + "
		|	" + Поле + ",";
	КонецЦикла;
	
	ТекстЗапроса =
		"ВЫБРАТЬ ПЕРВЫЕ " + Формат(Лимит, "ЧГ=") + ТекстПолейВыбора + "
		|	СУММА(" + Ресурс + ") КАК Значение
		|ИЗ
		|	РегистрСведений." + ИмяРегистра + "
		|" + УсловияОтбора.ТекстГДЕ + "
		|СГРУППИРОВАТЬ ПО " + СтрСоединить(ПоляГруппировки, ", ") + "
		|УПОРЯДОЧИТЬ ПО Значение " + Сортировка;
	
	Запрос = Новый Запрос(ТекстЗапроса);
	УстановитьПараметрыОтбора(Запрос, УсловияОтбора.Параметры);
	
	Результат = Запрос.Выполнить();
	Если Результат.Пустой() Тогда
		Возврат СформироватьОтветОшибки("no_data", "Данные за указанный период не найдены");
	КонецЕсли;
	
	ТаблицаРезультата = Результат.Выгрузить();
	Данные = Новый Массив;
	Для Каждого СтрокаТЗ Из ТаблицаРезультата Цикл
		Запись = Новый Структура;
		Для Каждого Поле Из ПоляГруппировки Цикл
			Запись.Вставить(Поле, ЗначениеДляJSON(СтрокаТЗ[Поле]));
		КонецЦикла;
		Запись.Вставить("Значение", СтрокаТЗ.Значение);
		Данные.Добавить(Запись);
	КонецЦикла;
	
	Возврат СформироватьОтветУспех(Данные, , ТекстЗапроса);

// compare — сравнение двух значений одного измерения
Функция ВыполнитьCompare(ИмяРегистра, Параметры)
	
	Ресурс = Параметры.resource;
	Лимит = ПолучитьЛимит(Параметры);
	ИзмерениеСравнения = Параметры.compare_by;
	Значения = Параметры.values;
	
	// Убираем compare_by из filters чтобы не дублировать в ГДЕ
	УсловияОтбора = ПостроитьУсловияОтбора(Параметры, , ИзмерениеСравнения);
	
	ТекстЗапроса =
		"ВЫБРАТЬ ПЕРВЫЕ " + Формат(Лимит, "ЧГ=") + "
		|	" + ИзмерениеСравнения + ",
		|	СУММА(" + Ресурс + ") КАК Значение
		|ИЗ
		|	РегистрСведений." + ИмяРегистра + "
		|ГДЕ
		|	" + ИзмерениеСравнения + " В (&ЗначенияСравнения)" +
		?(Не ПустаяСтрока(УсловияОтбора.ТекстУсловий), "
		|	И " + УсловияОтбора.ТекстУсловий, "") + "
		|СГРУППИРОВАТЬ ПО " + ИзмерениеСравнения;
	
	Запрос = Новый Запрос(ТекстЗапроса);
	
	МассивЗначений = Новый Массив;
	Для Каждого Элемент Из Значения Цикл
		МассивЗначений.Добавить(Элемент);
	КонецЦикла;
	Запрос.УстановитьПараметр("ЗначенияСравнения", МассивЗначений);
	УстановитьПараметрыОтбора(Запрос, УсловияОтбора.Параметры);
	
	Результат = Запрос.Выполнить();
	Если Результат.Пустой() Тогда
		Возврат СформироватьОтветОшибки("no_data", "Данные для сравнения не найдены");
	КонецЕсли;
	
	ТаблицаРезультата = Результат.Выгрузить();
	Данные = Новый Массив;
	ЗначенияПоИзмерению = Новый Соответствие;
	Для Каждого СтрокаТЗ Из ТаблицаРезультата Цикл
		ЗначениеИзмерения = СтрокаТЗ[ИзмерениеСравнения];
		Запись = Новый Структура;
		Запись.Вставить(ИзмерениеСравнения, ЗначениеДляJSON(ЗначениеИзмерения));
		Запись.Вставить("Значение", СтрокаТЗ.Значение);
		Данные.Добавить(Запись);
		ЗначенияПоИзмерению.Вставить(XMLСтрока(ЗначениеИзмерения), СтрокаТЗ.Значение);
	КонецЦикла;
	
	// Вычислить diff и percent
	Вычисляемые = Новый Структура;
	Знач1 = ЗначенияПоИзмерению.Получить(Значения[0]);
	Знач2 = ЗначенияПоИзмерению.Получить(Значения[1]);
	
	Если Знач1 <> Неопределено И Знач2 <> Неопределено Тогда
		Разница = Знач2 - Знач1;
		Вычисляемые.Вставить("diff", Разница);
		Если Знач1 <> 0 Тогда
			Вычисляемые.Вставить("percent", Окр(Разница / Знач1 * 100, 2));
		Иначе
			Вычисляемые.Вставить("percent", Неопределено);
		КонецЕсли;
	КонецЕсли;
	
	Возврат СформироватьОтветУспех(Данные, Вычисляемые, ТекстЗапроса);
	
КонецФункции

// ratio — отношение двух показателей
Функция ВыполнитьRatio(ИмяРегистра, Параметры)
	
	Ресурс = Параметры.resource;
	Лимит = ПолучитьЛимит(Параметры);
	Числитель = Параметры.numerator;
	Знаменатель = Параметры.denominator;
	
	// Убираем "Показатель" из filters
	УсловияОтбора = ПостроитьУсловияОтбора(Параметры, , "Показатель");
	
	ТекстЗапроса =
		"ВЫБРАТЬ ПЕРВЫЕ " + Формат(Лимит, "ЧГ=") + "
		|	Показатель,
		|	СУММА(" + Ресурс + ") КАК Значение
		|ИЗ
		|	РегистрСведений." + ИмяРегистра + "
		|ГДЕ
		|	Показатель В (&ЗначенияПоказателей)" +
		?(Не ПустаяСтрока(УсловияОтбора.ТекстУсловий), "
		|	И " + УсловияОтбора.ТекстУсловий, "") + "
		|СГРУППИРОВАТЬ ПО Показатель";
	
	Запрос = Новый Запрос(ТекстЗапроса);
	
	МассивПоказателей = Новый Массив;
	МассивПоказателей.Добавить(Числитель);
	МассивПоказателей.Добавить(Знаменатель);
	Запрос.УстановитьПараметр("ЗначенияПоказателей", МассивПоказателей);
	УстановитьПараметрыОтбора(Запрос, УсловияОтбора.Параметры);
	
	Результат = Запрос.Выполнить();
	Если Результат.Пустой() Тогда
		Возврат СформироватьОтветОшибки("no_data", "Данные для расчёта отношения не найдены");
	КонецЕсли;
	
	Данные = Новый Массив;
	ЗначенияПоказателей = Новый Соответствие;
	Выборка = Результат.Выбрать();
	Пока Выборка.Следующий() Цикл
		ИмяПоказателя = XMLСтрока(Выборка.Показатель);
		Запись = Новый Структура;
		Запись.Вставить("Показатель", ИмяПоказателя);
		Запись.Вставить("Значение", Выборка.Значение);
		Данные.Добавить(Запись);
		ЗначенияПоказателей.Вставить(ИмяПоказателя, Выборка.Значение);
	КонецЦикла;
	
	// Вычислить ratio
	ЗначЧислителя = ЗначенияПоказателей.Получить(Числитель);
	ЗначЗнаменателя = ЗначенияПоказателей.Получить(Знаменатель);
	
	Если ЗначЗнаменателя = Неопределено ИЛИ ЗначЗнаменателя = 0 Тогда
		Возврат СформироватьОтветОшибки("no_data",
			"Невозможно вычислить отношение: знаменатель (" + Знаменатель + ") равен нулю или отсутствует");
	КонецЕсли;
	
	Вычисляемые = Новый Структура;
	Вычисляемые.Вставить("ratio", Окр(ЗначЧислителя / ЗначЗнаменателя, 4));
	
	Возврат СформироватьОтветУспех(Данные, Вычисляемые, ТекстЗапроса);

	Лимит = ПолучитьЛимит(Параметры);
	ПоляГруппировки = ПолучитьПоляГруппировки(Параметры);
	Сортировка = ПолучитьСортировку(Параметры);
	
	Оператор = Параметры.condition_operator;
	Порог = Параметры.condition_value;
	
	УсловияОтбора = ПостроитьУсловияОтбора(Параметры);
	
	// Оператор через switch/case — защита от инъекций
	ТекстОператора = ПолучитьТекстОператора(Оператор);
	Если ТекстОператора = Неопределено Тогда
		Возврат СформироватьОтветОшибки("invalid_params",
			"Недопустимый оператор: " + Оператор + ". Допустимые: >, <, >=, <=, =");
	КонецЕсли;
	
	// Формируем SELECT-поля группировки
	ТекстПолейВыбора = "";
	Для Каждого Поле Из ПоляГруппировки Цикл
		ТекстПолейВыбора = ТекстПолейВыбора + "
		|	" + Поле + ",";
	КонецЦикла;
	
	ТекстЗапроса =
		"ВЫБРАТЬ ПЕРВЫЕ " + Формат(Лимит, "ЧГ=") + ТекстПолейВыбора + "
		|	СУММА(" + Ресурс + ") КАК Значение
		|ИЗ
		|	РегистрСведений." + ИмяРегистра + "
		|" + УсловияОтбора.ТекстГДЕ + "
		|СГРУППИРОВАТЬ ПО " + СтрСоединить(ПоляГруппировки, ", ") + "
		|ИМЕЮЩИЕ СУММА(" + Ресурс + ") " + ТекстОператора + " &Порог
		|УПОРЯДОЧИТЬ ПО Значение " + Сортировка;
	
	Запрос = Новый Запрос(ТекстЗапроса);
	Запрос.УстановитьПараметр("Порог", Порог);
	УстановитьПараметрыОтбора(Запрос, УсловияОтбора.Параметры);
	
	Результат = Запрос.Выполнить();
	Если Результат.Пустой() Тогда
		Возврат СформироватьОтветОшибки("no_data",
			"Данные, удовлетворяющие условию, не найдены");
	КонецЕсли;
	
	ТаблицаРезультата = Результат.Выгрузить();
	Данные = Новый Массив;
	Для Каждого СтрокаТЗ Из ТаблицаРезультата Цикл
		Запись = Новый Структура;
		Для Каждого Поле Из ПоляГруппировки Цикл
			Запись.Вставить(Поле, ЗначениеДляJSON(СтрокаТЗ[Поле]));
		КонецЦикла;
		Запись.Вставить("Значение", СтрокаТЗ.Значение);
		Данные.Добавить(Запись);
	КонецЦикла;
	
	Возврат СформироватьОтветУспех(Данные, , ТекстЗапроса);

// Построить текст условий ГДЕ из filters + period
//
// Параметры:
//   Параметры            - Структура - params из JSON-запроса
//   ВключатьПериод       - Булево - включать условия по периоду (Ложь для time_series)
//   ИсключитьИзмерение   - Строка - измерение, которое не нужно включать в ГДЕ
//                                    (используется в compare и ratio)
//
// Возвращает Структуру:
//   ТекстГДЕ       - Строка - полный текст "ГДЕ ..." (или "" если нет условий)
//   ТекстУсловий   - Строка - условия без ключевого слова ГДЕ
//   Параметры       - Соответствие - {ИмяПараметра → Значение} для УстановитьПараметр
//
Функция ПостроитьУсловияОтбора(Параметры, ВключатьПериод = Истина, ИсключитьИзмерение = "")
	
	Условия = Новый Массив;
	ПараметрыЗапроса = Новый Соответствие;
	
	// Фильтры по измерениям
	Если ЕстьСвойство(Параметры, "filters") И Параметры.filters <> Неопределено Тогда
		Для Каждого КлючЗначение Из Параметры.filters Цикл
			ИмяИзмерения = КлючЗначение.Ключ;
			
			Если ИмяИзмерения = ИсключитьИзмерение Тогда
				Продолжить;
			КонецЕсли;
			
			ИмяПараметра = "Фильтр_" + ИмяИзмерения;
			Условия.Добавить(ИмяИзмерения + " = &" + ИмяПараметра);
			ПараметрыЗапроса.Вставить(ИмяПараметра, КлючЗначение.Значение);
		КонецЦикла;
	КонецЕсли;
	
	// Условия по периоду
	Если ВключатьПериод И ЕстьСвойство(Параметры, "period") И Параметры.period <> Неопределено Тогда
		Период = Параметры.period;
		
		Если ЕстьСвойство(Период, "year") И Период.year <> Неопределено Тогда
			Условия.Добавить("ГОД(Период) = &Год");
			ПараметрыЗапроса.Вставить("Год", Период.year);
		КонецЕсли;
		
		Если ЕстьСвойство(Период, "month") И Период.month <> Неопределено Тогда
			Условия.Добавить("МЕСЯЦ(Период) = &Месяц");
			ПараметрыЗапроса.Вставить("Месяц", Период.month);
		КонецЕсли;
	КонецЕсли;
	
	// Собрать результат
	ТекстУсловий = СтрСоединить(Условия, "
	|	И ");
	
	Результат = Новый Структура;
	Результат.Вставить("ТекстУсловий", ТекстУсловий);
	
	Если Не ПустаяСтрока(ТекстУсловий) Тогда
		Результат.Вставить("ТекстГДЕ", "ГДЕ
		|	" + ТекстУсловий);
	Иначе
		Результат.Вставить("ТекстГДЕ", "");
	КонецЕсли;
	
	Результат.Вставить("Параметры", ПараметрыЗапроса);
	
	Возврат Результат;

// Валидировать параметры запроса перед выполнением
//
// Возвращает Структуру:
//   Успех               - Булево
//   ТипОшибки           - Строка (если не успех)
//   Сообщение           - Строка (если не успех)
//   ДопустимыеЗначения  - Массив (если invalid_params)
//
Функция ВалидироватьПараметры(ИмяРегистра, ИмяИнструмента, Параметры)
	
	Результат = Новый Структура("Успех", Истина);
	
	// Проверка существования регистра
	МетаРегистр = Метаданные.РегистрыСведений.Найти(ИмяРегистра);
	Если МетаРегистр = Неопределено Тогда
		Результат.Успех = Ложь;
		Результат.Вставить("ТипОшибки", "invalid_params");
		Результат.Вставить("Сообщение", "Регистр сведений не найден: " + ИмяРегистра);
		Результат.Вставить("ДопустимыеЗначения", Неопределено);
		Возврат Результат;
	КонецЕсли;
	
	// Проверка инструмента
	ДопустимыеИнструменты = Новый Массив;
	ДопустимыеИнструменты.Добавить("aggregate");
	ДопустимыеИнструменты.Добавить("group_by");
	ДопустимыеИнструменты.Добавить("top_n");
	ДопустимыеИнструменты.Добавить("time_series");
	ДопустимыеИнструменты.Добавить("compare");
	ДопустимыеИнструменты.Добавить("ratio");
	ДопустимыеИнструменты.Добавить("filtered");
	
	Если ДопустимыеИнструменты.Найти(ИмяИнструмента) = Неопределено Тогда
		Результат.Успех = Ложь;
		Результат.Вставить("ТипОшибки", "invalid_params");
		Результат.Вставить("Сообщение", "Неизвестный инструмент: " + ИмяИнструмента);
		Результат.Вставить("ДопустимыеЗначения", ДопустимыеИнструменты);
		Возврат Результат;
	КонецЕсли;
	
	// Проверка обязательного resource
	Если НЕ ЕстьСвойство(Параметры, "resource") ИЛИ ПустаяСтрока(Параметры.resource) Тогда
		Результат.Успех = Ложь;
		Результат.Вставить("ТипОшибки", "missing_params");
		Результат.Вставить("Сообщение", "Не указан ресурс (resource)");
		Результат.Вставить("ДопустимыеЗначения", Неопределено);
		Возврат Результат;
	КонецЕсли;
	
	// Проверка ресурса: у РегистрСведений числовое поле может лежать
	// и в Ресурсах, и в Реквизитах — проверяем обе коллекции.
	РесурсНайден = Ложь;
	ДопустимыеРесурсы = Новый Массив;

	Для Каждого Р Из МетаРегистр.Ресурсы Цикл
		ДопустимыеРесурсы.Добавить(Р.Имя);
		Если Р.Имя = Параметры.resource Тогда
			РесурсНайден = Истина;
		КонецЕсли;
	КонецЦикла;

	Для Каждого Р Из МетаРегистр.Реквизиты Цикл
		ДопустимыеРесурсы.Добавить(Р.Имя);
		Если Р.Имя = Параметры.resource Тогда
			РесурсНайден = Истина;
		КонецЕсли;
	КонецЦикла;

	Если НЕ РесурсНайден Тогда
		Результат.Успех = Ложь;
		Результат.Вставить("ТипОшибки", "invalid_params");
		Результат.Вставить("Сообщение",
			"Неизвестный ресурс: " + Параметры.resource);
		Результат.Вставить("ДопустимыеЗначения", ДопустимыеРесурсы);
		Возврат Результат;
	КонецЕсли;
	
	// Проверка измерений в filters
	Если ЕстьСвойство(Параметры, "filters") И Параметры.filters <> Неопределено Тогда
		Для Каждого КлючЗначение Из Параметры.filters Цикл
			ИмяИзмерения = КлючЗначение.Ключ;
			Если МетаРегистр.Измерения.Найти(ИмяИзмерения) = Неопределено
				И МетаРегистр.Реквизиты.Найти(ИмяИзмерения) = Неопределено Тогда
				
				ДопустимыеИзмерения = Новый Массив;
				Для Каждого Изм Из МетаРегистр.Измерения Цикл
					ДопустимыеИзмерения.Добавить(Изм.Имя);
				КонецЦикла;
				Результат.Успех = Ложь;
				Результат.Вставить("ТипОшибки", "invalid_params");
				Результат.Вставить("Сообщение",
					"Неизвестное измерение: " + ИмяИзмерения);
				Результат.Вставить("ДопустимыеЗначения", ДопустимыеИзмерения);
				Возврат Результат;
			КонецЕсли;
		КонецЦикла;
	КонецЕсли;
	
	// Специфичные проверки по инструменту
	Если ИмяИнструмента = "compare" Тогда
		Если НЕ ЕстьСвойство(Параметры, "compare_by") ИЛИ ПустаяСтрока(Параметры.compare_by) Тогда
			Результат.Успех = Ложь;
			Результат.Вставить("ТипОшибки", "missing_params");
			Результат.Вставить("Сообщение", "Для compare необходим параметр compare_by");
			Результат.Вставить("ДопустимыеЗначения", Неопределено);
			Возврат Результат;
		КонецЕсли;
		Если НЕ ЕстьСвойство(Параметры, "values") ИЛИ Параметры.values.Количество() <> 2 Тогда
			Результат.Успех = Ложь;
			Результат.Вставить("ТипОшибки", "missing_params");
			Результат.Вставить("Сообщение",
				"Для compare необходим параметр values с двумя значениями");
			Результат.Вставить("ДопустимыеЗначения", Неопределено);
			Возврат Результат;
		КонецЕсли;
		
	ИначеЕсли ИмяИнструмента = "ratio" Тогда
		Если НЕ ЕстьСвойство(Параметры, "numerator") ИЛИ ПустаяСтрока(Параметры.numerator) Тогда
			Результат.Успех = Ложь;
			Результат.Вставить("ТипОшибки", "missing_params");
			Результат.Вставить("Сообщение", "Для ratio необходим параметр numerator");
			Результат.Вставить("ДопустимыеЗначения", Неопределено);
			Возврат Результат;
		КонецЕсли;
		Если НЕ ЕстьСвойство(Параметры, "denominator") ИЛИ ПустаяСтрока(Параметры.denominator) Тогда
			Результат.Успех = Ложь;
			Результат.Вставить("ТипОшибки", "missing_params");
			Результат.Вставить("Сообщение", "Для ratio необходим параметр denominator");
			Результат.Вставить("ДопустимыеЗначения", Неопределено);
			Возврат Результат;
		КонецЕсли;
		
	ИначеЕсли ИмяИнструмента = "filtered" Тогда
		Если НЕ ЕстьСвойство(Параметры, "condition_operator") Тогда
			Результат.Успех = Ложь;
			Результат.Вставить("ТипОшибки", "missing_params");
			Результат.Вставить("Сообщение", "Для filtered необходим параметр condition_operator");
			Результат.Вставить("ДопустимыеЗначения", Неопределено);
			Возврат Результат;
		КонецЕсли;
		Если НЕ ЕстьСвойство(Параметры, "condition_value") Тогда
			Результат.Успех = Ложь;
			Результат.Вставить("ТипОшибки", "missing_params");
			Результат.Вставить("Сообщение", "Для filtered необходим параметр condition_value");
			Результат.Вставить("ДопустимыеЗначения", Неопределено);
			Возврат Результат;
		КонецЕсли;
		
	ИначеЕсли ИмяИнструмента = "group_by" ИЛИ ИмяИнструмента = "top_n" Тогда
		Если НЕ ЕстьСвойство(Параметры, "group_by") Тогда
			Результат.Успех = Ложь;
			Результат.Вставить("ТипОшибки", "missing_params");
			Результат.Вставить("Сообщение",
				"Для " + ИмяИнструмента + " необходим параметр group_by");
			Результат.Вставить("ДопустимыеЗначения", Неопределено);
			Возврат Результат;
		КонецЕсли;
	КонецЕсли;
	
	Возврат Результат;

// Распознать один из whitelist-паттернов. Возвращает Структуру с полями
// Тип (health|probe|distinct) + Лимит, ИмяРегистра, Поле — или Неопределено.
Функция РаспознатьDiscoveryЗапрос(Знач Текст)
	
	// Health-check: точное совпадение
	Если Текст = "ВЫБРАТЬ ПЕРВЫЕ 1 1 КАК Тест" Тогда
		Инфо = Новый Структура;
		Инфо.Вставить("Тип", "health");
		Возврат Инфо;
	КонецЕсли;
	
	// Probe: "ВЫБРАТЬ ПЕРВЫЕ <N> * ИЗ <register>"
	ПрефиксProbe = "ВЫБРАТЬ ПЕРВЫЕ ";
	Если Лев(Текст, СтрДлина(ПрефиксProbe)) = ПрефиксProbe Тогда
		Хвост = Сред(Текст, СтрДлина(ПрефиксProbe) + 1);
		ПозицияГвоздя = СтрНайти(Хвост, " * ИЗ ");
		Если ПозицияГвоздя > 0 Тогда
			ЛимитСтр = Лев(Хвост, ПозицияГвоздя - 1);
			ИмяРегистра = СокрЛП(Сред(Хвост, ПозицияГвоздя + СтрДлина(" * ИЗ ")));
			Лимит = РаспарситьПоложительноеЧисло(ЛимитСтр, 1000);
			Если Лимит <> Неопределено Тогда
				Инфо = Новый Структура;
				Инфо.Вставить("Тип", "probe");
				Инфо.Вставить("Лимит", Лимит);
				Инфо.Вставить("ИмяРегистра", ИмяРегистра);
				Возврат Инфо;
			КонецЕсли;
		КонецЕсли;
	КонецЕсли;
	
	// Distinct: "ВЫБРАТЬ РАЗЛИЧНЫЕ ПЕРВЫЕ <N> <field> ИЗ <register>"
	ПрефиксDistinct = "ВЫБРАТЬ РАЗЛИЧНЫЕ ПЕРВЫЕ ";
	Если Лев(Текст, СтрДлина(ПрефиксDistinct)) = ПрефиксDistinct Тогда
		Хвост = Сред(Текст, СтрДлина(ПрефиксDistinct) + 1);
		Позиция1 = СтрНайти(Хвост, " ");
		Если Позиция1 > 0 Тогда
			ЛимитСтр = Лев(Хвост, Позиция1 - 1);
			Хвост2 = Сред(Хвост, Позиция1 + 1);
			Позиция2 = СтрНайти(Хвост2, " ИЗ ");
			Если Позиция2 > 0 Тогда
				Поле = СокрЛП(Лев(Хвост2, Позиция2 - 1));
				ИмяРегистра = СокрЛП(Сред(Хвост2, Позиция2 + СтрДлина(" ИЗ ")));
				Лимит = РаспарситьПоложительноеЧисло(ЛимитСтр, 1000);
				Если Лимит <> Неопределено Тогда
					Инфо = Новый Структура;
					Инфо.Вставить("Тип", "distinct");
					Инфо.Вставить("Лимит", Лимит);
					Инфо.Вставить("Поле", Поле);
					Инфо.Вставить("ИмяРегистра", ИмяРегистра);
					Возврат Инфо;
				КонецЕсли;
			КонецЕсли;
		КонецЕсли;
	КонецЕсли;
	
	Возврат Неопределено;

// Пересобрать запрос из доверенного шаблона — сырой текст клиента не используется.

	КонецЕсли;
	
	ЛимитТекст = Формат(Инфо.Лимит, "ЧГ=");
	
	Если Инфо.Тип = "probe" Тогда
		Возврат "ВЫБРАТЬ ПЕРВЫЕ " + ЛимитТекст + " * ИЗ " + ДанныеРегистра.ПолноеИмя;
	ИначеЕсли Инфо.Тип = "distinct" Тогда
		Если Не ЭтоИдентификатор(Инфо.Поле) Тогда
			ВызватьИсключение "Недопустимое имя поля: " + Инфо.Поле;
		КонецЕсли;
		Возврат "ВЫБРАТЬ РАЗЛИЧНЫЕ ПЕРВЫЕ " + ЛимитТекст + " " + Инфо.Поле
			+ " ИЗ " + ДанныеРегистра.ПолноеИмя;

// Разобрать "РегистрСведений.<Имя>" или "РегистрНакопления.<Имя>" и проверить
// что такой регистр действительно существует в метаданных конфигурации.
Функция РазобратьИмяРегистра(Знач Текст)
	
	Позиция = СтрНайти(Текст, ".");
	Если Позиция <= 0 Тогда
		Возврат Неопределено;
	КонецЕсли;

		Возврат Неопределено;
	КонецЕсли;
	
	Если Вид = "РегистрСведений" Тогда

			Возврат Неопределено;
		КонецЕсли;
	Иначе
		Возврат Неопределено;
	КонецЕсли;
	
	Результат = Новый Структура;

// Проверить что текст — валидный 1С-идентификатор (буква/подчёркивание,
